# Runtime dependencies for both platforms.
#
# A single file serves the Windows and Linux CI jobs: platform-specific deps are
# guarded with PEP 508 environment markers, so `pip install -r requirements.txt`
# pulls the right set on each runner (see .github/workflows/build.yml).

# ---- shared ----
# (No PySDL2 — rendering/input go through the vendored hand-rolled sdl3w ctypes
#  binding: Windows ships SDL3 DLLs in windows/sdl3w/dll; Linux loads the system
#  libSDL3 / libSDL3_ttf.)
Pillow
pynput
hidapi
PyYAML
pystray

# ---- Windows only ----
# vgamepad is vendored under windows/vgamepad because upstream publishes only
# an sdist whose setup.py tries to launch the ViGEmBus MSI during CI installs.

# ---- Linux only ----
PyGObject<3.52 ; sys_platform == "linux" # gi / AppIndicator tray backend (linux/tray_linux.py)

# ---- build tool (used by CI to produce the release binaries) ----
pyinstaller
